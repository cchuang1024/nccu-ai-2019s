def solve_func(n, i, a, b, c):
    if i < n:
        for j in range(n):
            if j not in a and i+j not in b and i-j not in c:
                for solution in solve_func(n, i+1, a+[j], b+[i+j], c+[i-j]):
                    yield solution
    else:
        yield a

        
def solve_first(n):
    solutions = solve_func(n, 0, [], [], [])
    return next(solutions)
